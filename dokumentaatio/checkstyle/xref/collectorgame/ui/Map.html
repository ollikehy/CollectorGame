3   import java.awt.FlowLayout;
4   import java.awt.event.ActionEvent;
5   import java.awt.event.ActionListener;
6   import java.awt.image.BufferedImage;
7   import java.io.File;
8   import java.io.IOException;
9   import javax.imageio.ImageIO;
10  import javax.swing.ImageIcon;
11  import javax.swing.JButton;
12  import javax.swing.JFrame;
13  import javax.swing.JLabel;
14  
15  /**
16   * Luokassa luodaan karttaikkuna päävalikkoon.
17   * @author keolli
18   */
19  public class Map {
20  
21      JFrame frame;
22      
23      /**
24       * Konstruktori luo ikkunan mihini tuo lataa kartan ja lisää painikkeen.
25       * @throws IOException Heittää exceptionin jos tiedostoa ei löydy.
26       */
27      
28      public Map() throws IOException {
29          BufferedImage img = ImageIO.read(new File("src/main/resources/resources/screenshot.png"));
30          ImageIcon icon = new ImageIcon(img);
31          this.frame = new JFrame("Esimerkki kartta");
32          frame.setLayout(new FlowLayout());
33          frame.setSize(img.getWidth(), img.getHeight() + 60);
34          JLabel label = new JLabel();
35          label.setIcon(icon);
36          frame.add(label);
37          frame.setLocationRelativeTo(null);
38  
39          JButton button = new JButton();
40          button.setSize(img.getWidth(), 30);
41          button.setActionCommand("EXIT");
42          button.addActionListener(new ButtonClickListener());
43          button.setText("Main Menu");
44          frame.add(button);
45          
46          frame.setVisible(true);
47          frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
48      }
49  
50      private class ButtonClickListener implements ActionListener {
51  
52          @Override
53          public void actionPerformed(ActionEvent e) {
54              String command = e.getActionCommand();
55              if (command.equals("EXIT")) {
56                  frame.dispose();
57              }
58          }
59      }
60  
61  }
